\newcounter{chatlinenum}

%% Adjust text width to suit
\tikzset{chatstyle/.style={text width=4in,font=\sffamily,rounded corners=2pt,minimum height=1.6cm}}

%% Adjust width of minipage to suit, but greater than TikZ text width
\NewDocumentEnvironment{chat}{}{%
   \setcounter{chatlinenum}{0}
   \begin{minipage}{3.0in}
       \everypar={\chatline}
}{%
   \end{minipage}
}

\definecolor{mygreen}{HTML}{88EABB}

%% Alter colors to suit
\def\chatline#1\par{%
   \stepcounter{chatlinenum}%
   \noindent
   \ifodd\thechatlinenum
       \tikz[]{ \hspace{0.2\textwidth} \node[fill=lightgray,chatstyle]{\strut#1\strut}; 
       \node[chatstyle] (img) at (-1.5,0) {\includegraphics[height=1cm]{images/user1.png}}; }%
   \else
       \hfill
       \hspace{0.8\textwidth}
       \tikz[]{\node[fill=mygreen,chatstyle,align=left]{\strut#1\unskip\strut}; \node[chatstyle] (img) at (-1.5,0) {\includegraphics[height=1cm]{images/user2.png}};}%
   \fi
   \par
   \smallskip
}
